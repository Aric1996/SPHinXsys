template<int PKG_SIZE, int ADDRS_SIZE>
class SPH::BaseDataPackage< PKG_SIZE, ADDRS_SIZE >

Abstract base class for a data package which is given by a small mesh patch. note tha ADDRS_SIZE = PKG_SIZE + 2 * pkg_addrs_buffer_;.
