258 
260  int middle_level = ((int)total_levels - 1) / 2;
261  Real zero_level_cell_spacing = reference_cell_spacing * powern(2.0, middle_level);
262  cell_spacing_levels_.push_back(zero_level_cell_spacing);
263  MeshLevelType* zero_level_mesh
264  = new MeshLevelType(lower_bound, upper_bound, zero_level_cell_spacing, buffer_width);
265  mesh_levels_.push_back(zero_level_mesh);
266  Vecu zero_level_number_of_cells = zero_level_mesh->NumberOfCells();
268  this->copyMeshProperties(zero_level_mesh);
269 
271  for (size_t level = 1; level != total_levels; ++level) {
272  Real cell_spacing = zero_level_cell_spacing * powern(0.5, (int)level);
273  cell_spacing_levels_.push_back(cell_spacing);
274  Vecu number_of_cells = zero_level_number_of_cells * powern(2, (int)level);
275  MeshLevelType* mesh_level
276  = new MeshLevelType(lower_bound, number_of_cells, cell_spacing);
277  mesh_levels_.push_back(mesh_level);
278  }
279  };
280  virtual ~MultilevelMesh() {};
281  };
