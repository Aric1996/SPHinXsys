142  protected:

144  StdVec<MeshLevelType*> mesh_levels_;
145  public:
146  MultilevelMesh(MeshCompositionType& mesh_composition, ParticleAdaptation& particle_adaptation,
147  BoundingBox tentative_bounds, Real reference_spacing,
148  size_t total_levels, Real maximum_spacing_ratio, size_t buffer_width) :
149  BaseMeshType(mesh_composition, particle_adaptation, tentative_bounds,
150  reference_spacing, buffer_width), total_levels_(total_levels)
151  {
152  Real zero_level_spacing = reference_spacing * maximum_spacing_ratio;
153  for (size_t level = 0; level != total_levels_; ++level) {
154  Real spacing_level = zero_level_spacing * powerN(0.5, (int)level);
156  MeshLevelType* mesh_level =
157  new MeshLevelType(mesh_composition, particle_adaptation, tentative_bounds, spacing_level, buffer_width);
158  mesh_levels_.push_back(mesh_level);
159  }
160  };
161

165  };
