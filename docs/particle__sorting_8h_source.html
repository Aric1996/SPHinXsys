46  inline size_t median_of_three(const RandomAccessIterator& array, size_t l, size_t m, size_t r) const {
47  return comp_(array[l], array[m]) ? (comp_(array[m], array[r]) ? m : (comp_(array[l], array[r]) ? r : l))
48  : (comp_(array[r], array[m]) ? m : (comp_(array[r], array[l]) ? r : l));
49  }

51  inline size_t PseudoMedianOfNine(const RandomAccessIterator& array, const QuickSortParticleRange& range) const {
52  size_t offset = range.size_ / 8u;
53  return median_of_three(array,
54  median_of_three(array, 0, offset, offset * 2),
55  median_of_three(array, offset * 3, offset * 4, offset * 5),
56  median_of_three(array, offset * 6, offset * 7, range.size_ - 1));

60  size_t splitRange(QuickSortParticleRange& range) {
61  RandomAccessIterator array = range.begin_;
62  RandomAccessIterator key0 = range.begin_;
63  size_t m = PseudoMedianOfNine(array, range);
64  if (m) swap_particle_data_(array, array + m);
65 
66  size_t i = 0;
67  size_t j = range.size_;
68  // Partition interval [i+1,j-1] with key *key0.
69  for (;;) {
70  __TBB_ASSERT(i < j, NULL);
71  // Loop must terminate since array[l]==*key0.
72  do {
73  --j;
74  __TBB_ASSERT(i <= j, "bad ordering relation?");
75  } while (comp_(*key0, array[j]));
76  do {
77  __TBB_ASSERT(i <= j, NULL);
78  if (i == j) goto quick_sort_particle_partition;
79  ++i;
80  } while (comp_(array[i], *key0));
81  if (i == j) goto quick_sort_particle_partition;
82  swap_particle_data_(array + i, array + j);
83  }
84  quick_sort_particle_partition:
85  // Put the partition key were it belongs
86  swap_particle_data_(array + j, key0);
87  // array[l..j) is less or equal to key.
88  // array(j..r) is greater or equal to key.
89  // array[j] is equal to key
90  i = j + 1;
91  size_t new_range_size = range.size_ - i;
92  range.size_ = j;
93  return new_range_size;
94  }

125  template <typename RandomAccessIterator, typename Compare, typename SwapType>
126  RandomAccessIterator Partition(RandomAccessIterator first, RandomAccessIterator last, Compare& compare, SwapType& swap_particle_data)
127  {
128  auto pivot = std::prev(last, 1);
129  auto i = first;
130  for (auto j = first; j != pivot; ++j) {
131  // bool format
132  if (compare(*j, *pivot)) {
133  swap_particle_data(i++, j);
134  }
135  }
136  swap_particle_data(i, pivot);
137  return i;
138  }

141  void SerialQuickSort(RandomAccessIterator first, RandomAccessIterator last, Compare& compare, SwapType& swap_particle_data)
142  {
143  if (std::distance(first, last) > 1) {
144  RandomAccessIterator bound = Partition(first, last, compare, swap_particle_data);
145  SerialQuickSort(first, bound, compare, swap_particle_data);
146  SerialQuickSort(bound + 1, last, compare, swap_particle_data);

155  template< typename RandomAccessIterator, typename Compare, typename SwapType>
156  void InsertionSort(RandomAccessIterator First, RandomAccessIterator Last, Compare& compare, SwapType& swap_particle_data)
157  {
158  RandomAccessIterator min = First;
159  for (RandomAccessIterator i = First + 1; i < Last; ++i)
160  if (compare(*i, *min)) min = i;
161 
162  swap_particle_data(First, min);
163  while (++First < Last)
164  for (RandomAccessIterator j = First; compare(*j, *(j - 1)); --j)
165  swap_particle_data((j - 1), j);
166  }
