206  struct UpdateAReactionSpecies
207  {
208  Real operator () (Real input, Real production_rate, Real loss_rate, Real dt) const
209  {
210  return input * exp(-loss_rate * dt) + production_rate * (1.0 - exp(-loss_rate * dt)) / (loss_rate + TinyReal);
211  };
