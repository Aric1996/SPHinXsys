147  void getDiffusionChangeRate(size_t particle_i, size_t particle_j, Vecd& e_ij, Real surface_area_ij)
148  {
149  for (size_t m = 0; m < species_diffusion_.size(); ++m)
150  {
151  Real diff_coff_ij = species_diffusion_[m]->getInterParticleDiffusionCoff(particle_i, particle_j, e_ij);
152  size_t l = species_diffusion_[m]->gradient_species_index_;
153  Real phi_ij = species_n_[l][particle_i] - species_n_[l][particle_j];
154  diffusion_dt_[m][particle_i] += diff_coff_ij * phi_ij * surface_area_ij;
155  }
156  };
157 
163  virtual void updateSpeciesDiffusion(size_t particle_i, Real dt)
164  {
165  for (size_t m = 0; m < species_diffusion_.size(); ++m)
166  {
167  size_t k = species_diffusion_[m]->diffusion_species_index_;
168  species_n_[k][particle_i] += dt * diffusion_dt_[m][particle_i];
169  }
170  };
171 
172  virtual void Interaction(size_t index_i, Real dt = 0.0) override
173  {
174  DiffusionReactionParticles<BaseParticlesType, BaseMaterialType>* particles = this->particles_;
175  Neighborhood& inner_neighborhood = this->inner_configuration_[index_i];
176 
177  initializeDiffusionChangeRate(index_i);
178  for (size_t n = 0; n != inner_neighborhood.current_size_; ++n)
179  {
180  size_t index_j = inner_neighborhood.j_[n];
181  Real dW_ij_ = inner_neighborhood.dW_ij_[n];
182  Real r_ij_ = inner_neighborhood.r_ij_[n];
183  Vecd& e_ij = inner_neighborhood.e_ij_[n];
184 
185  const Vecd& gradi_ij = particles->getKernelGradient(index_i, index_j, dW_ij_, e_ij);
186  Real area_ij = 2.0 * Vol_[index_j] * dot(gradi_ij, e_ij) / r_ij_;
187  getDiffusionChangeRate(index_i, index_j, e_ij, area_ij);
188  }
189  };
190 
191  virtual void Update(size_t index_i, Real dt = 0.0) override
192  {
193  updateSpeciesDiffusion(index_i, dt);
194  };
195  public:
196  RelaxationOfAllDiffussionSpeciesInner(SPHBodyInnerRelation* body_inner_relation)
197  : InteractionDynamicsWithUpdate(body_inner_relation->sph_body_),
198  DiffusionReactionDataDelegateInner<BodyType, BaseParticlesType, BaseMaterialType>(body_inner_relation),
