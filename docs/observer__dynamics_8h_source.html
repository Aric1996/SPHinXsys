70  {

73 
74  for (size_t k = 0; k < this->contact_configuration_.size(); ++k)
75  {
76  StdLargeVec<Real>& Vol_k = *(contact_Vol_[k]);
77  StdLargeVec<VariableType>& data_k = *(contact_data_[k]);
78  Neighborhood& contact_neighborhood = (*this->contact_configuration_[k])[index_i];
79  for (size_t n = 0; n != contact_neighborhood.current_size_; ++n)
80  {
81  size_t index_j = contact_neighborhood.j_[n];
82  Real weight_j = contact_neighborhood.W_ij_[n] * Vol_k[index_j];
83 
84  observed_quantity += weight_j * data_k[index_j];
85  ttl_weight += weight_j;
86  }
87  }
88  interpolated_quantities_[index_i] = observed_quantity / (ttl_weight + TinyReal);
89  };
