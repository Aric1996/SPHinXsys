55  class ObservingAQuantity :
56  public InteractionDynamics, public ObservationDataDelegate<TargetParticlesType>
57  {
58  protected:
60  StdLargeVec<DataType> observed_quantities_;
61  StdVec<StdLargeVec<Real>*> contact_Vol_;
62  StdVec<StdLargeVec<DataType>*> contact_data_;
63 
64  virtual void Interaction(size_t index_i, Real dt = 0.0) override
65  {
66  DataType observed_quantity(0);
67  Real ttl_weight(0);
68 
69  for (size_t k = 0; k < this->contact_configuration_.size(); ++k)
70  {
71  StdLargeVec<Real>& Vol_k = *(contact_Vol_[k]);
72  StdLargeVec<DataType>& data_k = *(contact_data_[k]);
73  Neighborhood& contact_neighborhood
74  = this->contact_configuration_[k][index_i];
75  for (size_t n = 0; n != contact_neighborhood.current_size_; ++n)
76  {
77  size_t index_j = contact_neighborhood.j_[n];
78  Real weight_j = contact_neighborhood.W_ij_[n] * Vol_k[index_j];
79 
80  observed_quantity += weight_j * data_k[index_j];
81  ttl_weight += weight_j;
82  }
83  }
84  observed_quantities_[index_i] = observed_quantity / (ttl_weight + TinyReal);
85  };
86  public:

92  {
