107  virtual void Interaction(size_t index_i, Real dt = 0.0) override
108  {
109  Vecd& dvel_dt_ave_i = dvel_dt_ave_[index_i];
110  Real Vol_i = Vol_[index_i];
111  Vecd& vel_ave_i = vel_ave_[index_i];
112  Vecd& n_i = n_[index_i];
113 
114  Vecd force(0);
115  for (size_t k = 0; k < contact_configuration_.size(); ++k)
116  {
117  StdLargeVec<Real>& Vol_k = *(contact_Vol_[k]);
118  StdLargeVec<Real>& rho_n_k = *(contact_rho_n_[k]);
119  StdLargeVec<Real>& p_k = *(contact_p_[k]);
120  StdLargeVec<Vecd>& vel_n_k = *(contact_vel_n_[k]);
121  StdLargeVec<Vecd>& dvel_dt_others_k = *(contact_dvel_dt_others_[k]);
122  Fluid* fluid_k = contact_material_[k];
123  RiemannSolverType* riemann_solver_k = riemann_solvers_[k];
124  Neighborhood& contact_neighborhood = (*contact_configuration_[k])[index_i];
125  for (size_t n = 0; n != contact_neighborhood.current_size_; ++n)
126  {
127  size_t index_j = contact_neighborhood.j_[n];
128  Vecd e_ij = contact_neighborhood.e_ij_[n];
129  Real r_ij = contact_neighborhood.r_ij_[n];
130  Real face_wall_external_acceleration
131  = dot((dvel_dt_others_k[index_j] - dvel_dt_ave_i), e_ij);
132  Real p_in_wall = p_k[index_j] + rho_n_k[index_j] * r_ij * SMAX(0.0, face_wall_external_acceleration);
133  Real rho_in_wall = fluid_k->DensityFromPressure(p_in_wall);
134 
135  FluidState state_l(rho_n_k[index_j], vel_n_k[index_j], p_k[index_j]);
136  FluidState state_r(rho_in_wall, vel_ave_i, p_in_wall);
137  Real p_star = riemann_solver_k->getPStar(state_l, state_r, n_i);
138  force -= 2.0 * p_star * e_ij * Vol_i * Vol_k[index_j] * contact_neighborhood.dW_ij_[n];
139  }
140  }
141  force_from_fluid_[index_i] = viscous_force_from_fluid_[index_i] + force;
142  };
